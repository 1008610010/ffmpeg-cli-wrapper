001package net.bramp.ffmpeg.progress;
002
003import com.google.common.net.InetAddresses;
004
005import java.io.IOException;
006import java.net.InetAddress;
007import java.net.URI;
008import java.net.URISyntaxException;
009import java.util.concurrent.CountDownLatch;
010
011import static com.google.common.base.Preconditions.checkNotNull;
012
013public abstract class AbstractSocketProgressParser implements ProgressParser {
014
015  final StreamProgressParser parser;
016
017  Thread thread; // Thread for handling incoming connections
018
019  public AbstractSocketProgressParser(ProgressListener listener) {
020    this.parser = new StreamProgressParser(listener);
021  }
022
023  /**
024   * Creates a URL to parse to FFmpeg based on the scheme, address and port.
025   *
026   * TODO Move this method to somewhere better.
027   *
028   * @param scheme
029   * @param address
030   * @param port
031   * @return
032   * @throws URISyntaxException
033   */
034  static URI createUri(String scheme, InetAddress address, int port) throws URISyntaxException {
035    checkNotNull(address);
036    return new URI(scheme, null /* userInfo */, InetAddresses.toUriString(address), port,
037        null /* path */, null /* query */, null /* fragment */);
038  }
039
040  protected abstract String getThreadName();
041
042  protected abstract Runnable getRunnable(CountDownLatch startSignal);
043
044  /**
045   *
046   * @exception IllegalThreadStateException if the parser was already started.
047   */
048  public synchronized void start() {
049    if (thread != null) {
050      throw new IllegalThreadStateException("Parser already started");
051    }
052
053    String name = getThreadName() + "(" + getUri().toString() + ")";
054
055    CountDownLatch startSignal = new CountDownLatch(1);
056    Runnable runnable = getRunnable(startSignal);
057
058    thread = new Thread(runnable, name);
059    thread.start();
060
061    // Block until the thread has started
062    try {
063      startSignal.await();
064    } catch (InterruptedException e) {
065      Thread.currentThread().interrupt();
066    }
067  }
068
069  @Override
070  public void stop() throws IOException {
071    if (thread != null) {
072      thread.interrupt(); // This unblocks processStream();
073
074      try {
075        thread.join();
076      } catch (InterruptedException e) {
077        Thread.currentThread().interrupt();
078      }
079    }
080  }
081
082  @Override
083  public void close() throws IOException {
084    stop();
085  }
086}
