001package net.bramp.ffmpeg.progress;
002
003import com.google.common.base.MoreObjects;
004import net.bramp.ffmpeg.FFmpegUtils;
005import org.apache.commons.lang3.math.Fraction;
006
007import java.util.Objects;
008
009import static com.google.common.base.Preconditions.checkNotNull;
010
011/**
012 * TODO Change to be immutable
013 */
014public class Progress {
015  public long frame = 0;
016  public Fraction fps = Fraction.ZERO;
017
018  // TODO stream_0_0_q=0.0
019
020  public long bitrate = 0;
021  public long total_size = 0;
022  public long out_time_ms = 0;
023
024  public long dup_frames = 0;
025  public long drop_frames = 0;
026  public float speed = 0;
027  public String progress = "";
028
029  public Progress() {
030    // Nothing
031  }
032
033  public Progress(long frame, float fps, long bitrate, long total_size, long out_time_ms,
034      long dup_frames, long drop_frames, float speed, String progress) {
035    this.frame = frame;
036    this.fps = Fraction.getFraction(fps);
037    this.bitrate = bitrate;
038    this.total_size = total_size;
039    this.out_time_ms = out_time_ms;
040    this.dup_frames = dup_frames;
041    this.drop_frames = drop_frames;
042    this.speed = speed;
043    this.progress = progress;
044  }
045
046  /**
047   * Parses values from the line, into this object.
048   *
049   * @param line A single line of output from ffmpeg
050   * @return true if the record is finished
051   */
052  protected boolean parseLine(String line) {
053    line = checkNotNull(line).trim();
054    if (line.isEmpty()) {
055      return false; // Skip empty lines
056    }
057
058    final String[] args = line.split("=", 2);
059    if (args.length != 2) {
060      // invalid argument, so skip
061      return false;
062    }
063
064    final String key = checkNotNull(args[0]);
065    final String value = checkNotNull(args[1]);
066
067    switch (key) {
068      case "frame":
069        frame = Long.parseLong(value);
070        return false;
071      case "fps":
072        fps = Fraction.getFraction(value);
073        return false;
074      case "bitrate":
075        // TODO bitrate could be "N/A"
076        bitrate = FFmpegUtils.parseBitrate(value);
077        return false;
078      case "total_size":
079        // TODO could be "N/A"
080        total_size = Long.parseLong(value);
081        return false;
082      case "out_time_ms":
083        out_time_ms = Long.parseLong(value);
084        return false;
085      case "out_time":
086        // There is also out_time_ms, so we ignore out_time.
087        // TODO maybe in the future actually parse out_time, if a out_time_ms wasn't provided.
088        return false;
089      case "dup_frames":
090        dup_frames = Long.parseLong(value);
091        return false;
092      case "drop_frames":
093        drop_frames = Long.parseLong(value);
094        return false;
095      case "speed":
096        // TODO Could be "N/A"
097        speed = Float.parseFloat(value.replace("x", ""));
098        return false;
099      case "progress":
100        // TODO Is only "continue" or "end". Change to enum
101        // TODO After "end" stream is closed
102        progress = value;
103        return true; // The progress field is always last
104
105      default:
106        // TODO
107        // stream_%d_%d_q= file_index, index, quality
108        // stream_%d_%d_psnr_%c=%2.2f, file_index, index, type{Y, U, V}, quality // Enable with
109        // AV_CODEC_FLAG_PSNR
110        // stream_%d_%d_psnr_all
111        return false; // Ignore for the moment
112    }
113  }
114
115  public boolean isEnd() {
116    return Objects.equals(progress, "end");
117  }
118
119  @Override
120  public boolean equals(Object o) {
121    if (this == o)
122      return true;
123    if (o == null || getClass() != o.getClass())
124      return false;
125    Progress progress1 = (Progress) o;
126    return frame == progress1.frame && bitrate == progress1.bitrate
127        && total_size == progress1.total_size && out_time_ms == progress1.out_time_ms
128        && dup_frames == progress1.dup_frames && drop_frames == progress1.drop_frames
129        && Float.compare(progress1.speed, speed) == 0 && Objects.equals(fps, progress1.fps)
130        && Objects.equals(progress, progress1.progress);
131  }
132
133  @Override
134  public int hashCode() {
135    return Objects.hash(frame, fps, bitrate, total_size, out_time_ms, dup_frames, drop_frames,
136        speed, progress);
137  }
138
139  @Override
140  public String toString() {
141    return MoreObjects.toStringHelper(this)
142        .add("frame", frame)
143        .add("fps", fps)
144        .add("bitrate", bitrate)
145        .add("total_size", total_size)
146        .add("out_time_ms", out_time_ms)
147        .add("dup_frames", dup_frames)
148        .add("drop_frames", drop_frames)
149        .add("speed", speed)
150        .add("progress", progress)
151        .toString();
152  }
153}
