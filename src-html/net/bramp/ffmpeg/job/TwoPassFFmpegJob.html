001package net.bramp.ffmpeg.job;
002
003import com.google.common.base.Throwables;
004import net.bramp.ffmpeg.FFmpeg;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.io.File;
009import java.io.IOException;
010import java.nio.file.FileSystems;
011import java.nio.file.Files;
012import java.nio.file.Path;
013import java.util.List;
014
015public class TwoPassFFmpegJob extends FFmpegJob {
016
017  final static Logger LOG = LoggerFactory.getLogger(TwoPassFFmpegJob.class);
018
019  final String passlogPrefix;
020  final List<String> args1;
021  final List<String> args2;
022
023  public TwoPassFFmpegJob(FFmpeg ffmpeg, String passlogPrefix, List<String> args1,
024      List<String> args2) {
025    super(ffmpeg);
026    this.passlogPrefix = passlogPrefix;
027    this.args1 = args1;
028    this.args2 = args2;
029  }
030
031  protected void deletePassLog() throws IOException {
032    Path path = FileSystems.getDefault().getPath(passlogPrefix);
033
034    Files.deleteIfExists(path);
035  }
036
037  public void run() {
038    state = State.RUNNING;
039
040    try {
041      ffmpeg.run(args1);
042      ffmpeg.run(args2);
043
044      deletePassLog();
045
046      state = State.FINISHED;
047
048    } catch (Throwable t) {
049      state = State.FAILED;
050      Throwables.propagate(t);
051    }
052  }
053}
