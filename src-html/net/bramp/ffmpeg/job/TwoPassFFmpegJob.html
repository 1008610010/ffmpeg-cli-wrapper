001package net.bramp.ffmpeg.job;
002
003import com.google.common.base.Throwables;
004import net.bramp.ffmpeg.FFmpeg;
005import net.bramp.ffmpeg.builder.FFmpegBuilder;
006import net.bramp.ffmpeg.progress.ProgressListener;
007
008import javax.annotation.Nullable;
009import java.io.IOException;
010import java.nio.file.DirectoryStream;
011import java.nio.file.Files;
012import java.nio.file.Path;
013import java.nio.file.Paths;
014import java.util.UUID;
015
016import static com.google.common.base.Preconditions.checkNotNull;
017
018public class TwoPassFFmpegJob extends FFmpegJob {
019
020  final String passlogPrefix;
021  final FFmpegBuilder builder;
022
023  public TwoPassFFmpegJob(FFmpeg ffmpeg, FFmpegBuilder builder) {
024    this(ffmpeg, builder, null);
025  }
026
027  public TwoPassFFmpegJob(FFmpeg ffmpeg, FFmpegBuilder builder, @Nullable ProgressListener listener) {
028    super(ffmpeg, listener);
029
030    // Random prefix so multiple runs don't clash
031    this.passlogPrefix = UUID.randomUUID().toString();
032    this.builder = checkNotNull(builder).setPassPrefix(passlogPrefix);
033
034    // Build the args now (but throw away the results). This allows the illegal arguments to be
035    // caught early, but also allows the ffmpeg command to actually alter the arguments when
036    // running.
037    this.builder.setPass(1).build();
038  }
039
040  protected void deletePassLog() throws IOException {
041    final Path cwd = Paths.get("");
042    try (DirectoryStream<Path> stream = Files.newDirectoryStream(cwd, passlogPrefix + "*.log*")) {
043      for (Path p : stream) {
044        Files.deleteIfExists(p);
045      }
046    }
047  }
048
049  public void run() {
050    state = State.RUNNING;
051
052    try {
053      try {
054        // Two pass
055        final boolean override = builder.getOverrideOutputFiles();
056
057        FFmpegBuilder b1 = builder.setPass(1).overrideOutputFiles(true);
058        ffmpeg.run(b1, listener);
059
060        FFmpegBuilder b2 = builder.setPass(2).overrideOutputFiles(override);
061        ffmpeg.run(b2, listener);
062
063      } finally {
064        deletePassLog();
065      }
066      state = State.FINISHED;
067
068    } catch (Throwable t) {
069      state = State.FAILED;
070
071      Throwables.throwIfUnchecked(t);
072      throw new RuntimeException(t);
073    }
074  }
075}
