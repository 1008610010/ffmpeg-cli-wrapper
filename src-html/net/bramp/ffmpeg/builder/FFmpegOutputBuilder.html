083  /**
084   * Checks if the URI is valid for streaming to
085   * 
086   * @param uri
087   * @return
088   */
089  public static URI checkValidStream(URI uri) {
090    String scheme = checkNotNull(uri).getScheme();
091    scheme = checkNotNull(scheme, "URI is missing a scheme").toLowerCase();
092
093    if (rtps.contains(scheme)) {
094      return uri;
095    }
096
097    if (udpTcp.contains(scheme)) {
098      if (uri.getPort() == -1) {
099        throw new IllegalArgumentException("must set port when using udp or tcp scheme");
100      }
101      return uri;
102    }
103
104    throw new IllegalArgumentException("not a valid output URL, must use rtp/tcp/udp scheme");
105  }

463  public EncodingOptions buildOptions() {
464    // TODO When/if modelmapper supports @ConstructorProperties, we map this
465    // object, instead of doing new XXX(...)
466    // https://github.com/jhalterman/modelmapper/issues/44
467    return new EncodingOptions(new MainEncodingOptions(format, startOffset, duration),
468        new AudioEncodingOptions(audio_enabled, audio_codec, audio_channels, audio_sample_rate,
469            audio_bit_depth, audio_bit_rate, audio_quality), new VideoEncodingOptions(
470            video_enabled, video_codec, video_frame_rate, video_width, video_height,
471            video_bit_rate, video_frames, video_filter, video_preset));
472  }
473
474  protected List<String> build(int pass) {
475    Preconditions.checkState(parent != null, "Can not build without parent being set");
476
477    ImmutableList.Builder<String> args = new ImmutableList.Builder<String>();
478
479    if (targetSize > 0) {
480      checkState(parent.inputs.size() == 1, "Target size does not support multiple inputs");
481
482      String firstInput = parent.inputs.get(0);
483      FFmpegProbeResult input = parent.inputProbes.get(firstInput);
484
485      checkState(input != null, "Target size must be used with setInput(FFmpegProbeResult)");
486
487      // TODO factor in start time and/or number of frames
488
489      double durationInSeconds = input.format.duration;
490      long totalBitRate =
491          (long) Math.floor((targetSize * 8) / durationInSeconds) - pass_padding_bitrate;
492
493      // TODO Calculate audioBitRate
494
495      if (video_enabled && video_bit_rate == 0) {
496        // Video (and possibly audio)
497        long audioBitRate = audio_enabled ? audio_bit_rate : 0;
498        video_bit_rate = totalBitRate - audioBitRate;
499      } else if (audio_enabled && audio_bit_rate == 0) {
500        // Just Audio
501        audio_bit_rate = totalBitRate;
502      }
503    }
504
505    if (strict != FFmpegBuilder.Strict.NORMAL) {
506      args.add("-strict").add(strict.toString().toLowerCase());
507    }
508
509    if (!Strings.isNullOrEmpty(format)) {
510      args.add("-f").add(format);
511    }
512
513    if (startOffset != null) {
514      args.add("-ss").add(millisecondsToString(startOffset));
515    }
516
517    if (duration != null) {
518      args.add("-t").add(millisecondsToString(duration));
519    }
520
521    if (video_enabled) {
522
523      if (video_frames != null) {
524        args.add("-vframes").add(String.format("%d", video_frames));
525      }
526
527      if (!Strings.isNullOrEmpty(video_codec)) {
528        args.add("-vcodec").add(video_codec);
529      }
530
531      if (video_copyinkf) {
532        args.add("-copyinkf");
533      }
534
535      if (!Strings.isNullOrEmpty(video_movflags)) {
536        args.add("-movflags").add(video_movflags);
537      }
538
539      if (video_width != 0 && video_height != 0) {
540        args.add("-s").add(String.format("%dx%d", video_width, video_height));
541      }
542
543      if (video_frame_rate != null) {
544        // args.add("-r").add(String.format("%2f", video_frame_rate));
545        args.add("-r").add(video_frame_rate.toString());
546      }
547
548      if (video_bit_rate > 0) {
549        args.add("-b:v").add(String.format("%d", video_bit_rate));
550      }
551
552      if (!Strings.isNullOrEmpty(video_preset)) {
553        args.add("-vpre").add(video_preset);
554      }
555
556      if (!Strings.isNullOrEmpty(video_filter)) {
557        checkState(parent.inputs.size() == 1,
558            "Video filter only works with one input, instead use setVideoFilterComplex(..)");
559        args.add("-vf").add(video_filter);
560      }
561
562      if (!Strings.isNullOrEmpty(video_filter_complex)) {
563        args.add("-filter_complex").add(video_filter_complex);
564      }
565
566      if (!Strings.isNullOrEmpty(video_bit_stream_filter)) {
567        args.add("-bsf:v", video_bit_stream_filter);
568      }
569
570      for (String meta : video_meta_tags) {
571        args.add("-metadata").add("\"" + meta + "\"");
572      }
573
574    } else {
575      args.add("-vn");
576    }
577
578    if (audio_enabled && pass != 1) {
579      if (!Strings.isNullOrEmpty(audio_codec)) {
580        args.add("-acodec").add(audio_codec);
581      }
582
583      if (audio_channels > 0) {
584        args.add("-ac").add(String.format("%d", audio_channels));
585      }
586
587      if (audio_sample_rate > 0) {
588        args.add("-ar").add(String.format("%d", audio_sample_rate));
589      }
590
591      if (!Strings.isNullOrEmpty(audio_bit_depth)) {
592        args.add("-sample_fmt").add(audio_bit_depth);
593      }
594
595      if (audio_bit_rate > 0 && audio_quality > 0 && throwWarnings) {
596        // I'm not sure, but it seems audio_quality overrides
597        // audio_bit_rate, so don't allow both
598        throw new IllegalStateException("Only one of audio_bit_rate and audio_quality can be set");
599      }
600
601      if (audio_bit_rate > 0) {
602        args.add("-b:a").add(String.format("%d", audio_bit_rate));
603      }
604
605      if (audio_quality > 0) {
606        args.add("-aq").add(String.format("%d", audio_quality));
607      }
608
609      if (!Strings.isNullOrEmpty(audio_bit_stream_filter)) {
610        args.add("-bsf:a", audio_bit_stream_filter);
611      }
612
613    } else {
614      args.add("-an");
615    }
616
617    if (!subtitle_enabled)
618      args.add("-sn");
619
620    args.addAll(extra_args);
621
622    if (filename != null && uri != null) {
623      throw new IllegalStateException("Only one of filename and uri can be set");
624    }
625
626    // Output
627    if (pass == 1) {
628      args.add(DEVNULL);
629    } else if (filename != null) {
630      args.add(filename);
631    } else if (uri != null) {
632      args.add(uri.toString());
633    } else {
634      assert (false);
635    }
636
637    return args.build();
638  }
