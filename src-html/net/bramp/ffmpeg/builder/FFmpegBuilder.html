001package net.bramp.ffmpeg.builder;
002
003import com.google.common.base.Preconditions;
004import com.google.common.collect.ImmutableList;
005import net.bramp.ffmpeg.FFmpegUtils;
006import net.bramp.ffmpeg.probe.FFmpegProbeResult;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.util.ArrayList;
011import java.util.List;
012import java.util.Map;
013import java.util.TreeMap;
014import java.util.concurrent.TimeUnit;
015
016import static com.google.common.base.Preconditions.checkNotNull;
017
018/**
019 * Builds a ffmpeg command line
020 *
021 * @author bramp
022 */
023public class FFmpegBuilder {
024
025  final static Logger LOG = LoggerFactory.getLogger(FFmpegBuilder.class);
026
027  public enum Strict {
028    VERY, // strictly conform to a older more strict version of the spec or reference software
029    STRICT, // strictly conform to all the things in the spec no matter what consequences
030    NORMAL, // normal
031    UNOFFICAL, // allow unofficial extensions
032    EXPERIMENTAL;
033
034    // ffmpeg command line requires these options in the lower case
035    public String toString() {
036      return name().toLowerCase();
037    }
038  }
039
040  // Global Settings
041  boolean override = true;
042  int pass = 0;
043  String pass_prefix;
044
045  // Input settings
046  Long startOffset; // in millis
047  final List<String> inputs = new ArrayList<>();
048  final Map<String, FFmpegProbeResult> inputProbes = new TreeMap<>();
049
050  // Output
051  final List<FFmpegOutputBuilder> outputs = new ArrayList<FFmpegOutputBuilder>();
052
053  public FFmpegBuilder overrideOutputFiles(boolean override) {
054    this.override = override;
055    return this;
056  }
057
058  public boolean getOverrideOutputFiles() {
059    return this.override;
060  }
061
062  public FFmpegBuilder setPass(int pass) {
063    this.pass = pass;
064    return this;
065  }
066
067  public FFmpegBuilder setPassPrefix(String prefix) {
068    this.pass_prefix = prefix;
069    return this;
070  }
071
072  public FFmpegBuilder addInput(FFmpegProbeResult result) {
073    checkNotNull(result);
074    String filename = checkNotNull(result.format).filename;
075    inputProbes.put(filename, result);
076    return addInput(filename);
077  }
078
079  public FFmpegBuilder addInput(String filename) {
080    checkNotNull(filename);
081    inputs.add(filename);
082    return this;
083  }
084
085  protected void clearInputs() {
086    inputs.clear();
087    inputProbes.clear();
088  }
089
090  public FFmpegBuilder setInput(FFmpegProbeResult result) {
091    checkNotNull(result);
092    clearInputs();
093    return addInput(result);
094  }
095
096  public FFmpegBuilder setInput(String filename) {
097    checkNotNull(filename);
098    clearInputs();
099    return addInput(filename);
100  }
101
102  public FFmpegBuilder setStartOffset(long duration, TimeUnit units) {
103    checkNotNull(duration);
104    checkNotNull(units);
105
106    this.startOffset = units.toMillis(duration);
107
108    return this;
109  }
110
111  /**
112   * Create new output file
113   * 
114   * @param filename
115   * @return A new FFmpegOutputBuilder
116   */
117  public FFmpegOutputBuilder addOutput(String filename) {
118    FFmpegOutputBuilder output = new FFmpegOutputBuilder(this, filename);
119    outputs.add(output);
120    return output;
121  }
122
123  /**
124   * Create new output (to stdout)
125   */
126  public FFmpegOutputBuilder addStreamedOutput() {
127    return addOutput("-");
128  }
129
130  public List<String> build() {
131    ImmutableList.Builder<String> args = new ImmutableList.Builder<String>();
132
133    Preconditions.checkArgument(!inputs.isEmpty(), "At least one input must be specified");
134    Preconditions.checkArgument(!outputs.isEmpty(), "At least one output must be specified");
135
136    args.add(override ? "-y" : "-n");
137    args.add("-v", "error"); // TODO make configurable
138
139    if (startOffset != null) {
140      args.add("-ss").add(FFmpegUtils.millisecondsToString(startOffset));
141    }
142
143    for (String input : inputs) {
144      args.add("-i").add(input);
145    }
146
147    if (pass > 0) {
148      args.add("-pass").add(Integer.toString(pass));
149
150      if (pass_prefix != null) {
151        args.add("-passlogfile").add(pass_prefix);
152      }
153    }
154
155    for (FFmpegOutputBuilder output : this.outputs) {
156      args.addAll(output.build(pass));
157    }
158
159    return args.build();
160  }
161}
