001package net.bramp.ffmpeg.builder;
002
003import com.google.common.base.Preconditions;
004import com.google.common.collect.ImmutableList;
005import net.bramp.ffmpeg.FFmpegUtils;
006import net.bramp.ffmpeg.probe.FFmpegProbeResult;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.net.URI;
011import java.util.ArrayList;
012import java.util.List;
013import java.util.Map;
014import java.util.TreeMap;
015import java.util.concurrent.TimeUnit;
016
017import static com.google.common.base.Preconditions.checkArgument;
018import static com.google.common.base.Preconditions.checkNotNull;
019
020/**
021 * Builds a ffmpeg command line
022 *
023 * @author bramp
024 */
025public class FFmpegBuilder {
026
027  final static Logger LOG = LoggerFactory.getLogger(FFmpegBuilder.class);
028
029  public enum Strict {
030    VERY, // strictly conform to a older more strict version of the specifications or reference
031          // software
032    STRICT, // strictly conform to all the things in the specificiations no matter what consequences
033    NORMAL, // normal
034    UNOFFICAL, // allow unofficial extensions
035    EXPERIMENTAL;
036
037    // ffmpeg command line requires these options in lower case
038    @Override
039    public String toString() {
040      return name().toLowerCase();
041    }
042  }
043
044  /**
045   * Log level options: https://ffmpeg.org/ffmpeg.html#Generic-options
046   */
047  public enum Verbosity {
048    QUIET, PANIC, FATAL, ERROR, WARNING, INFO, VERBOSE, DEBUG;
049
050    @Override
051    public String toString() {
052      return name().toLowerCase();
053    }
054  }
055
056  // Global Settings
057  boolean override = true;
058  int pass = 0;
059  String pass_directory = "";
060  String pass_prefix;
061  Verbosity verbosity = Verbosity.ERROR;
062  URI progress;
063  String user_agent;
064
065  // Input settings
066  String format;
067  Long startOffset; // in millis
068  boolean read_at_native_frame_rate = false;
069  final List<String> inputs = new ArrayList<>();
070  final Map<String, FFmpegProbeResult> inputProbes = new TreeMap<>();
071
072  final List<String> extra_args = new ArrayList<>();
073
074  // Output
075  final List<FFmpegOutputBuilder> outputs = new ArrayList<>();
076
077  public FFmpegBuilder overrideOutputFiles(boolean override) {
078    this.override = override;
079    return this;
080  }
081
082  public boolean getOverrideOutputFiles() {
083    return this.override;
084  }
085
086  public FFmpegBuilder setPass(int pass) {
087    this.pass = pass;
088    return this;
089  }
090
091  public FFmpegBuilder setPassDirectory(String directory) {
092    this.pass_directory = checkNotNull(directory);
093    return this;
094  }
095
096  public FFmpegBuilder setPassPrefix(String prefix) {
097    this.pass_prefix = checkNotNull(prefix);
098    return this;
099  }
100
101  public FFmpegBuilder setVerbosity(Verbosity verbosity) {
102    checkNotNull(verbosity);
103    this.verbosity = verbosity;
104    return this;
105  }
106
107  public FFmpegBuilder setUserAgent(String userAgent) {
108    this.user_agent = checkNotNull(userAgent);
109    return this;
110  }
111
112  public FFmpegBuilder readAtNativeFrameRate() {
113    this.read_at_native_frame_rate = true;
114    return this;
115  }
116
117  public FFmpegBuilder addInput(FFmpegProbeResult result) {
118    checkNotNull(result);
119    String filename = checkNotNull(result.format).filename;
120    inputProbes.put(filename, result);
121    return addInput(filename);
122  }
123
124  public FFmpegBuilder addInput(String filename) {
125    checkNotNull(filename);
126    inputs.add(filename);
127    return this;
128  }
129
130  protected void clearInputs() {
131    inputs.clear();
132    inputProbes.clear();
133  }
134
135  public FFmpegBuilder setInput(FFmpegProbeResult result) {
136    clearInputs();
137    return addInput(result);
138  }
139
140  public FFmpegBuilder setInput(String filename) {
141    clearInputs();
142    return addInput(filename);
143  }
144
145  public FFmpegBuilder setFormat(String format) {
146    this.format = checkNotNull(format);
147    return this;
148  }
149
150  public FFmpegBuilder setStartOffset(long duration, TimeUnit units) {
151    checkNotNull(duration);
152    checkNotNull(units);
153
154    this.startOffset = units.toMillis(duration);
155
156    return this;
157  }
158
159  public FFmpegBuilder addProgress(URI uri) {
160    this.progress = checkNotNull(uri);
161    return this;
162  }
163
164
165  /**
166   * Add additional ouput arguments (for flags which aren't currently supported).
167   *
168   * @param values The extra arguments.
169   * @return this
170   */
171  public FFmpegBuilder addExtraArgs(String... values) {
172    checkArgument(values.length > 0, "One or more values must be supplied");
173    for (String value : values) {
174      extra_args.add(checkNotNull(value));
175    }
176    return this;
177  }
178
179
180  /**
181   * Adds new output file.
182   *
183   * @param filename output file path
184   * @return A new {@link FFmpegOutputBuilder}
185   */
186  public FFmpegOutputBuilder addOutput(String filename) {
187    FFmpegOutputBuilder output = new FFmpegOutputBuilder(this, filename);
188    outputs.add(output);
189    return output;
190  }
191
192  /**
193   * Adds new output file.
194   *
195   * @param uri output file uri typically a stream
196   * @return A new {@link FFmpegOutputBuilder}
197   */
198  public FFmpegOutputBuilder addOutput(URI uri) {
199    FFmpegOutputBuilder output = new FFmpegOutputBuilder(this, uri);
200    outputs.add(output);
201    return output;
202  }
203
204  /**
205   * Adds an existing FFmpegOutputBuilder. This is similar to calling the other addOuput methods but
206   * instead allows an existing FFmpegOutputBuilder to be used, and reused.
207   *
208   * <pre>
209   * <code>List&lt;String&gt; args = new FFmpegBuilder()
210   *   .addOutput(new FFmpegOutputBuilder()
211   *     .setFilename(&quot;output.flv&quot;)
212   *     .setVideoCodec(&quot;flv&quot;)
213   *   )
214   *   .build();</code>
215   * </pre>
216   *
217   * @param output FFmpegOutputBuilder to add
218   * @return this
219   */
220  public FFmpegBuilder addOutput(FFmpegOutputBuilder output) {
221    outputs.add(output);
222    return this;
223  }
224
225  /**
226   * Create new output (to stdout)
227   * 
228   * @return A new {@link FFmpegOutputBuilder}
229   */
230  public FFmpegOutputBuilder addStdoutOutput() {
231    return addOutput("-");
232  }
233
234  public List<String> build() {
235    ImmutableList.Builder<String> args = new ImmutableList.Builder<String>();
236
237    Preconditions.checkArgument(!inputs.isEmpty(), "At least one input must be specified");
238    Preconditions.checkArgument(!outputs.isEmpty(), "At least one output must be specified");
239
240    args.add(override ? "-y" : "-n");
241    args.add("-v", this.verbosity.toString());
242
243    if (user_agent != null) {
244      args.add("-user-agent", user_agent);
245    }
246
247    if (startOffset != null) {
248      args.add("-ss", FFmpegUtils.millisecondsToString(startOffset));
249    }
250
251    if (format != null) {
252      args.add("-f", format);
253    }
254
255    if (read_at_native_frame_rate) {
256      args.add("-re");
257    }
258
259    if (progress != null) {
260      args.add("-progress", progress.toString());
261    }
262
263    args.addAll(extra_args);
264
265    for (String input : inputs) {
266      args.add("-i", input);
267    }
268
269    if (pass > 0) {
270      args.add("-pass", Integer.toString(pass));
271
272      if (pass_prefix != null) {
273        args.add("-passlogfile", pass_directory + pass_prefix);
274      }
275    }
276
277    for (FFmpegOutputBuilder output : this.outputs) {
278      args.addAll(output.build(this, pass));
279    }
280
281    return args.build();
282  }
283}
