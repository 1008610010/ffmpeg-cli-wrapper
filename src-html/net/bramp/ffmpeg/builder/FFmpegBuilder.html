001package net.bramp.ffmpeg.builder;
002
003import com.google.common.base.Preconditions;
004import com.google.common.collect.ImmutableList;
005import net.bramp.ffmpeg.FFmpegUtils;
006import net.bramp.ffmpeg.probe.FFmpegProbeResult;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.net.URI;
011import java.util.ArrayList;
012import java.util.List;
013import java.util.Map;
014import java.util.TreeMap;
015import java.util.concurrent.TimeUnit;
016
017import static com.google.common.base.Preconditions.checkArgument;
018import static com.google.common.base.Preconditions.checkNotNull;
019
020/**
021 * Builds a ffmpeg command line
022 *
023 * @author bramp
024 */
025public class FFmpegBuilder {
026
027  final static Logger LOG = LoggerFactory.getLogger(FFmpegBuilder.class);
028
029  public enum Strict {
030    VERY, // strictly conform to a older more strict version of the spec or reference software
031    STRICT, // strictly conform to all the things in the spec no matter what consequences
032    NORMAL, // normal
033    UNOFFICAL, // allow unofficial extensions
034    EXPERIMENTAL;
035
036    // ffmpeg command line requires these options in the lower case
037    public String toString() {
038      return name().toLowerCase();
039    }
040  }
041
042  // Global Settings
043  boolean override = true;
044  int pass = 0;
045  String pass_prefix;
046
047  // Input settings
048  String format;
049  Long startOffset; // in millis
050  final List<String> inputs = new ArrayList<>();
051  final Map<String, FFmpegProbeResult> inputProbes = new TreeMap<>();
052
053  final List<String> extra_args = new ArrayList<>();
054
055  // Output
056  final List<FFmpegOutputBuilder> outputs = new ArrayList<FFmpegOutputBuilder>();
057
058  public FFmpegBuilder overrideOutputFiles(boolean override) {
059    this.override = override;
060    return this;
061  }
062
063  public boolean getOverrideOutputFiles() {
064    return this.override;
065  }
066
067  public FFmpegBuilder setPass(int pass) {
068    this.pass = pass;
069    return this;
070  }
071
072  public FFmpegBuilder setPassPrefix(String prefix) {
073    this.pass_prefix = prefix;
074    return this;
075  }
076
077  public FFmpegBuilder addInput(FFmpegProbeResult result) {
078    checkNotNull(result);
079    String filename = checkNotNull(result.format).filename;
080    inputProbes.put(filename, result);
081    return addInput(filename);
082  }
083
084  public FFmpegBuilder addInput(String filename) {
085    checkNotNull(filename);
086    inputs.add(filename);
087    return this;
088  }
089
090  protected void clearInputs() {
091    inputs.clear();
092    inputProbes.clear();
093  }
094
095  public FFmpegBuilder setInput(FFmpegProbeResult result) {
096    checkNotNull(result);
097    clearInputs();
098    return addInput(result);
099  }
100
101  public FFmpegBuilder setInput(String filename) {
102    checkNotNull(filename);
103    clearInputs();
104    return addInput(filename);
105  }
106
107  public FFmpegBuilder setFormat(String format) {
108    this.format = checkNotNull(format);
109    return this;
110  }
111
112  public FFmpegBuilder setStartOffset(long duration, TimeUnit units) {
113    checkNotNull(duration);
114    checkNotNull(units);
115
116    this.startOffset = units.toMillis(duration);
117
118    return this;
119  }
120
121  /**
122   * Add additional ouput arguments (for flags which aren't currently supported).
123   *
124   * @param values
125   */
126  public FFmpegBuilder addExtraArgs(String... values) {
127    checkArgument(values.length > 0, "One or more values must be supplied");
128    for (String value : values) {
129      extra_args.add(checkNotNull(value));
130    }
131    return this;
132  }
133
134
135  /**
136   * Create new output file
137   * 
138   * @param filename
139   * @return A new FFmpegOutputBuilder
140   */
141  public FFmpegOutputBuilder addOutput(String filename) {
142    FFmpegOutputBuilder output = new FFmpegOutputBuilder(this, filename);
143    outputs.add(output);
144    return output;
145  }
146
147  public FFmpegOutputBuilder addOutput(URI uri) {
148    FFmpegOutputBuilder output = new FFmpegOutputBuilder(this, uri);
149    outputs.add(output);
150    return output;
151  }
152
153  /**
154   * Create new output (to stdout)
155   */
156  public FFmpegOutputBuilder addStdoutOutput() {
157    return addOutput("-");
158  }
159
160  public List<String> build() {
161    ImmutableList.Builder<String> args = new ImmutableList.Builder<String>();
162
163    Preconditions.checkArgument(!inputs.isEmpty(), "At least one input must be specified");
164    Preconditions.checkArgument(!outputs.isEmpty(), "At least one output must be specified");
165
166    args.add(override ? "-y" : "-n");
167    args.add("-v", "error"); // TODO make configurable
168
169    if (startOffset != null) {
170      args.add("-ss").add(FFmpegUtils.millisecondsToString(startOffset));
171    }
172
173    if (format != null) {
174      args.add("-f", format);
175    }
176
177    args.addAll(extra_args);
178
179    for (String input : inputs) {
180      args.add("-i").add(input);
181    }
182
183    if (pass > 0) {
184      args.add("-pass").add(Integer.toString(pass));
185
186      if (pass_prefix != null) {
187        args.add("-passlogfile").add(pass_prefix);
188      }
189    }
190
191    for (FFmpegOutputBuilder output : this.outputs) {
192      args.addAll(output.build(pass));
193    }
194
195    return args.build();
196  }
197}
