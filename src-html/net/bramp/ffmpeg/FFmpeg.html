062  final static Pattern CODECS_REGEX = Pattern
063      .compile("^ ([ D][ E][VAS][ S][ D][ T]) (\\S+)\\s+(.*)$");

103  public FFmpeg(@Nonnull String path, ProcessFunction runFunction) throws IOException {
104    Preconditions.checkArgument(!Strings.isNullOrEmpty(path));
105    this.runFunc = checkNotNull(runFunction);
106    this.path = path;
107    this.version = version();
108  }
109
110  private BufferedReader wrapInReader(Process p) {
111    return new BufferedReader(new InputStreamReader(p.getInputStream(), Charsets.UTF_8));
112  }
113
114  public synchronized @Nonnull String version() throws IOException {
115    String version;
116    Process p = runFunc.run(ImmutableList.of(path, "-version"));
117    try {
118      BufferedReader r = wrapInReader(p);
119      version = r.readLine();
120      IOUtils.copy(r, new NullOutputStream()); // Throw away rest of the output
121      FFmpegUtils.throwOnError("ffmpeg", p);
122    } finally {
123      p.destroy();
124    }
125
126    return version;
127  }
128
129  public synchronized @Nonnull List<Codec> codecs() throws IOException {
130    if (this.codecs == null) {
131      codecs = new ArrayList<Codec>();
132
133      Process p = runFunc.run(ImmutableList.of(path, "-codecs"));
134      try {
135        BufferedReader r = wrapInReader(p);
136        String line;
137        while ((line = r.readLine()) != null) {
138          Matcher m = CODECS_REGEX.matcher(line);
139          if (!m.matches())
140            continue;
141
142          codecs.add(new Codec(m.group(2), m.group(3), m.group(1)));
143        }
144
145        FFmpegUtils.throwOnError("ffmpeg", p);
146        this.codecs = ImmutableList.copyOf(codecs);
147      } finally {
148        p.destroy();
149      }
150
151    }
152
153    return codecs;
154  }
155
156
157  public synchronized @Nonnull List<Format> formats() throws IOException {
158    if (this.formats == null) {
159      formats = new ArrayList<Format>();
160
161      Process p = runFunc.run(ImmutableList.of(path, "-formats"));
162      try {
163        BufferedReader r = wrapInReader(p);
164        String line;
165        while ((line = r.readLine()) != null) {
166          Matcher m = FORMATS_REGEX.matcher(line);
167          if (!m.matches())
168            continue;
169
170          formats.add(new Format(m.group(2), m.group(3), m.group(1)));
171        }
172
173        FFmpegUtils.throwOnError("ffmpeg", p);
174        this.formats = ImmutableList.copyOf(formats);
175      } finally {
176        p.destroy();
177      }
178    }
179    return formats;
180  }
181
182  public void run(List<String> args) throws IOException {
183    List<String> newArgs = ImmutableList.<String>builder().add(path).addAll(args).build();
184
185    Process p = runFunc.run(newArgs);
186    try {
187      // Now block reading ffmpeg's stdout. We are effectively throwing away the output.
188      IOUtils.copy(wrapInReader(p), System.out); // TODO Should I be outputting to stdout?
189
190      FFmpegUtils.throwOnError("ffmpeg", p);
191
192    } finally {
193      p.destroy();
194    }
195  }
