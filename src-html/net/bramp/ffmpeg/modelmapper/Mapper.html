001package net.bramp.ffmpeg.modelmapper;
002
003import net.bramp.ffmpeg.builder.FFmpegOutputBuilder;
004import net.bramp.ffmpeg.options.AudioEncodingOptions;
005import net.bramp.ffmpeg.options.EncodingOptions;
006import net.bramp.ffmpeg.options.MainEncodingOptions;
007import net.bramp.ffmpeg.options.VideoEncodingOptions;
008import org.modelmapper.ModelMapper;
009import org.modelmapper.TypeMap;
010import org.modelmapper.config.Configuration;
011import org.modelmapper.convention.NameTokenizers;
012
013import static net.bramp.ffmpeg.modelmapper.NotDefaultCondition.notDefault;
014
015/**
016 * Copies values from one type of object to another
017 *
018 * @author bramp
019 */
020public class Mapper {
021
022  final private static ModelMapper mapper = newModelMapper();
023
024  private static <S, D> TypeMap<S, D> createTypeMap(ModelMapper mapper, Class<S> sourceType,
025      Class<D> destinationType, Configuration config) {
026
027    return mapper.createTypeMap(sourceType, destinationType, config)
028    // We setPropertyCondition because ModelMapper seems to ignore this in
029    // the config
030        .setPropertyCondition(config.getPropertyCondition());
031  }
032
033  private static ModelMapper newModelMapper() {
034    final ModelMapper mapper = new ModelMapper();
035
036    Configuration config =
037        mapper.getConfiguration().copy().setFieldMatchingEnabled(true)
038            .setPropertyCondition(notDefault).setSourceNameTokenizer(NameTokenizers.UNDERSCORE);
039
040    createTypeMap(mapper, MainEncodingOptions.class, FFmpegOutputBuilder.class, config);
041    createTypeMap(mapper, AudioWrapper.class, FFmpegOutputBuilder.class, config);
042    createTypeMap(mapper, VideoWrapper.class, FFmpegOutputBuilder.class, config);
043
044    return mapper;
045  }
046
047  /**
048   * Simple wrapper object, to inject the word "audio" in the property name
049   */
050  static class AudioWrapper {
051    public final AudioEncodingOptions audio;
052
053    AudioWrapper(AudioEncodingOptions audio) {
054      this.audio = audio;
055    }
056  }
057
058  /**
059   * Simple wrapper object, to inject the word "video" in the property name
060   */
061  static class VideoWrapper {
062    public final VideoEncodingOptions video;
063
064    VideoWrapper(VideoEncodingOptions video) {
065      this.video = video;
066    }
067  }
068
069  public static void map(MainEncodingOptions opts, FFmpegOutputBuilder dest) {
070    mapper.map(opts, dest);
071  }
072
073  public static void map(AudioEncodingOptions opts, FFmpegOutputBuilder dest) {
074    mapper.map(new AudioWrapper(opts), dest);
075  }
076
077  public static void map(VideoEncodingOptions opts, FFmpegOutputBuilder dest) {
078    mapper.map(new VideoWrapper(opts), dest);
079  }
080
081  public static void map(EncodingOptions opts, FFmpegOutputBuilder dest) {
082    map(opts.getMain(), dest);
083
084    if (opts.getAudio().enabled) {
085      map(opts.getAudio(), dest);
086    }
087    if (opts.getVideo().enabled) {
088      map(opts.getVideo(), dest);
089    }
090  }
091
092}
