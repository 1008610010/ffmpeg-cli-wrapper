001package net.bramp.ffmpeg.nut;
002
003import com.google.common.base.MoreObjects;
004
005import java.io.IOException;
006
007public class PacketHeader {
008
009  long startcode;
010  long forwardPtr;
011  int checksum; // header checksum
012
013  long end; // End byte of packet
014
015  public void read(NutDataInputStream in, long startcode) throws IOException {
016    this.startcode = startcode;
017    forwardPtr = in.readVarLong();
018    if (forwardPtr > 4096) {
019      long expected = in.getCRC();
020      checksum = in.readInt();
021      if (checksum != expected) {
022        // TODO This code path has never been tested.
023        throw new IOException(String.format("invalid header checksum %X want %X", expected,
024            checksum));
025      }
026    }
027
028    in.resetCRC();
029    end = in.offset() + forwardPtr - 4; // 4 bytes for footer CRC
030  }
031
032  @Override
033  public String toString() {
034    MoreObjects.ToStringHelper helper =
035        MoreObjects.toStringHelper(this).add("startcode", Packet.Startcode.toString(startcode))
036            .add("forwardPtr", forwardPtr);
037
038    if (forwardPtr > 4096) {
039      helper = helper.add("checksum", checksum);
040    }
041    return helper.toString();
042  }
043}
