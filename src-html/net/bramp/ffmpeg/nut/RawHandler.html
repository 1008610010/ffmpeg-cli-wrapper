018  private static int[] bytesToInts(byte[] bytes) {
019    IntBuffer buf = ByteBuffer.wrap(bytes).order(ByteOrder.BIG_ENDIAN).asIntBuffer();
020
021    int[] data = new int[buf.capacity()];
022    buf.get(data);
023    return data;
024  }

049  /**
050   * Parses a FourCC into a AudioEncoding based on the following rules:<br>
051   * "ALAW" = A-LAW<br>
052   * "ULAW" = MU-LAW<br>
053   * P[type][interleaving][bits] = little-endian PCM<br>
054   * [bits][interleaving][type]P = big-endian PCM<br>
055   * Where:<br>
056   * &nbsp;&nbsp;[type] is S for signed integer, U for unsigned integer, F for IEEE float<br>
057   * &nbsp;&nbsp;[interleaving] is D for default, P is for planar.<br>
058   * &nbsp;&nbsp;[bits] is 8/16/24/32<br>
059   *
060   * @param header The stream's header.
061   * @return The AudioFormat matching this header.
062   */
063  public static AudioFormat streamToAudioFormat(final StreamHeaderPacket header) {
064    checkNotNull(header);
065    checkArgument(header.type == StreamHeaderPacket.AUDIO);
066
067    // Vars that go into the AudioFormat
068    AudioFormat.Encoding encoding;
069    float sampleRate = header.sampleRate.floatValue();
070    int bits = 8;
071    boolean bigEndian = false;
072
073    final byte[] fourcc = header.fourcc;
074
075    if (Arrays.equals(fourcc, new byte[] {'A', 'L', 'A', 'W'})) {
076      encoding = AudioFormat.Encoding.ALAW;
077
078    } else if (Arrays.equals(fourcc, new byte[] {'U', 'L', 'A', 'W'})) {
079      encoding = AudioFormat.Encoding.ULAW;
080
081    } else if (fourcc.length == 4) {
082      byte type;
083      byte interleaving;
084
085      if (fourcc[0] == 'P') {
086        bigEndian = false;
087        type = fourcc[1];
088        interleaving = fourcc[2];
089        bits = fourcc[3];
090      } else if (fourcc[3] == 'P') {
091        bigEndian = true;
092        type = fourcc[2];
093        interleaving = fourcc[1];
094        bits = fourcc[0];
095      } else {
096        throw new IllegalArgumentException("unknown fourcc value: " + fourcc);
097      }
098
099      if (interleaving != 'D') {
100        throw new IllegalArgumentException("unsupported interleaving '" + interleaving
101            + "' in fourcc value " + fourcc);
102      }
103
104      switch (type) {
105        case 'S':
106          encoding = AudioFormat.Encoding.PCM_SIGNED;
107          break;
108        case 'U':
109          encoding = AudioFormat.Encoding.PCM_UNSIGNED;
110          break;
111        case 'F':
112          encoding = AudioFormat.Encoding.PCM_FLOAT;
113          break;
114        default:
115          throw new IllegalArgumentException("unknown fourcc '" + fourcc + "' type: " + type);
116      }
117    } else {
118      throw new IllegalArgumentException("unknown fourcc value: " + fourcc);
119    }
120
121    int frameSize = (bits * header.channels) / 8;
122    float frameRate = sampleRate; // This may not be true for the compressed formats
123
124    return new AudioFormat(encoding, sampleRate, bits, header.channels, frameSize, frameRate,
125        bigEndian);
126  }
127
128  public static AudioInputStream toAudioInputStream(Frame frame) {
129    checkNotNull(frame);
130    final StreamHeaderPacket header = checkNotNull(frame.stream.header);
131    checkArgument(header.type == StreamHeaderPacket.AUDIO);
132
133    AudioFormat format = streamToAudioFormat(header);
134    InputStream stream = new ByteArrayInputStream(frame.data);
135
136    return new AudioInputStream(stream, format, frame.data.length / format.getFrameSize());
137  }
