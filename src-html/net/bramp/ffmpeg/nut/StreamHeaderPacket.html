001package net.bramp.ffmpeg.nut;
002
003import com.google.common.base.MoreObjects;
004import org.apache.commons.lang3.math.Fraction;
005
006import java.io.IOException;
007
008public class StreamHeaderPacket extends Packet {
009
010  public final static int VIDEO = 0;
011  public final static int AUDIO = 1;
012  public final static int SUBTITLE = 2;
013  public final static int USER_DATA = 3;
014
015  int id;
016  long type; // One of VIDEO/AUDIO/SUBTITLE/USER_DATA // TODO Convert to enum.
017  byte[] fourcc;
018  int timeBaseId;
019  int msbPtsShift;
020  int maxPtsDistance;
021  long decodeDelay;
022  long flags;
023  byte[] codecSpecificData;
024
025  // If video
026  int width;
027  int height;
028  int sampleWidth;
029  int sampleHeight;
030  long colorspaceType;
031
032  // If audio
033  Fraction sampleRate = Fraction.ZERO;
034  int channels;
035
036  protected void readBody(NutDataInputStream in) throws IOException {
037
038    id = in.readVarInt();
039    type = in.readVarLong();
040    fourcc = in.readVarArray();
041
042    if (fourcc.length != 2 && fourcc.length != 4) {
043      // TODO In future fourcc could be a different size, but for sanity checking lets leave this
044      // check in.
045      throw new IOException("Unexpected fourcc length: " + fourcc.length);
046    }
047
048    timeBaseId = in.readVarInt();
049    msbPtsShift = in.readVarInt();
050    if (msbPtsShift >= 16) {
051      throw new IOException("invalid msbPtsShift " + msbPtsShift + " want < 16");
052    }
053    maxPtsDistance = in.readVarInt();
054    decodeDelay = in.readVarLong();
055    flags = in.readVarLong();
056    codecSpecificData = in.readVarArray();
057
058    if (type == VIDEO) {
059      width = in.readVarInt();
060      height = in.readVarInt();
061
062      if (width == 0 || height == 0) {
063        throw new IOException("invalid video dimensions " + width + "x" + height);
064      }
065
066      sampleWidth = in.readVarInt();
067      sampleHeight = in.readVarInt();
068
069      // Both MUST be 0 if unknown otherwise both MUST be nonzero.
070      if ((sampleWidth == 0 || sampleHeight == 0) && sampleWidth != sampleHeight) {
071        throw new IOException("invalid video sample dimensions " + sampleWidth + "x" + sampleHeight);
072      }
073
074      colorspaceType = in.readVarLong();
075
076    } else if (type == AUDIO) {
077      int samplerateNum = in.readVarInt();
078      int samplerateDenom = in.readVarInt();
079      sampleRate = Fraction.getFraction(samplerateNum, samplerateDenom);
080      channels = in.readVarInt();
081    }
082  }
083
084  @Override
085  public String toString() {
086    return MoreObjects.toStringHelper(this).add("header", header).add("id", id).add("type", type)
087        .add("fourcc", new String(fourcc)).add("timeBaseId", timeBaseId)
088        .add("msbPtsShift", msbPtsShift).add("maxPtsDistance", maxPtsDistance)
089        .add("decodeDelay", decodeDelay).add("flags", flags)
090        .add("codecSpecificData", codecSpecificData).add("width", width).add("height", height)
091        .add("sampleWidth", sampleWidth).add("sampleHeight", sampleHeight)
092        .add("colorspaceType", colorspaceType).add("sampleRate", sampleRate)
093        .add("channels", channels).add("footer", footer).toString();
094  }
095}
