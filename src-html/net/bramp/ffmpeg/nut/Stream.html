001package net.bramp.ffmpeg.nut;
002
003import org.apache.commons.lang3.math.Fraction;
004
005import java.io.IOException;
006
007public class Stream {
008  final StreamHeaderPacket header;
009
010  final Fraction timeBase;
011  long last_pts = 0;
012
013  public Stream(MainHeaderPacket header, StreamHeaderPacket streamHeader) throws IOException {
014    this.header = streamHeader;
015    if (streamHeader.timeBaseId >= header.timeBase.length) {
016      throw new IOException("Invalid timeBaseId " + streamHeader.timeBaseId + " must be < "
017          + header.timeBase.length);
018    }
019    this.timeBase = header.timeBase[streamHeader.timeBaseId];
020  }
021}
