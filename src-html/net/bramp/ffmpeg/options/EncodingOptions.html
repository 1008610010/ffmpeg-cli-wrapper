001package net.bramp.ffmpeg.options;
002
003import java.beans.ConstructorProperties;
004
005/**
006 * 
007 * @author bramp
008 *
009 */
010public class EncodingOptions {
011
012  public final MainEncodingOptions main;
013  public final AudioEncodingOptions audio;
014  public final VideoEncodingOptions video;
015
016  @ConstructorProperties({"main", "audio", "video"})
017  public EncodingOptions(MainEncodingOptions main, AudioEncodingOptions audio,
018      VideoEncodingOptions video) {
019    this.main = main;
020    this.audio = audio;
021    this.video = video;
022  }
023
024  public MainEncodingOptions getMain() {
025    return main;
026  }
027
028  public AudioEncodingOptions getAudio() {
029    return audio;
030  }
031
032  public VideoEncodingOptions getVideo() {
033    return video;
034  }
035}
