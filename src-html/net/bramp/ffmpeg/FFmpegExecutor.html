001package net.bramp.ffmpeg;
002
003import net.bramp.ffmpeg.builder.FFmpegBuilder;
004import net.bramp.ffmpeg.job.FFmpegJob;
005import net.bramp.ffmpeg.job.SinglePassFFmpegJob;
006import net.bramp.ffmpeg.job.TwoPassFFmpegJob;
007import net.bramp.ffmpeg.progress.ProgressListener;
008
009import java.io.IOException;
010
011import static com.google.common.base.Preconditions.checkNotNull;
012
013public class FFmpegExecutor {
014
015  final FFmpeg ffmpeg;
016  final FFprobe ffprobe;
017
018  public FFmpegExecutor() throws IOException {
019    this(new FFmpeg(), new FFprobe());
020  }
021
022  public FFmpegExecutor(FFmpeg ffmpeg) throws IOException {
023    this(ffmpeg, new FFprobe());
024  }
025
026  public FFmpegExecutor(FFmpeg ffmpeg, FFprobe ffprobe) {
027    this.ffmpeg = checkNotNull(ffmpeg);
028    this.ffprobe = checkNotNull(ffprobe);
029  }
030
031  public FFmpegJob createJob(FFmpegBuilder builder) {
032    return new SinglePassFFmpegJob(ffmpeg, builder);
033  }
034
035  public FFmpegJob createJob(FFmpegBuilder builder, ProgressListener listener) {
036    return new SinglePassFFmpegJob(ffmpeg, builder, listener);
037  }
038
039  /**
040   * Creates a two pass job, which will execute FFmpeg twice to produce a better quality output.
041   * More info: https://trac.ffmpeg.org/wiki/x264EncodingGuide#twopass
042   * 
043   * @param builder The FFmpegBuilder
044   * @return A new two-pass FFmpegJob
045   */
046  public FFmpegJob createTwoPassJob(FFmpegBuilder builder) {
047    return new TwoPassFFmpegJob(ffmpeg, builder);
048  }
049}
