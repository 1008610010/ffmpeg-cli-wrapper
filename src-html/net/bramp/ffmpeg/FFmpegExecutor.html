001package net.bramp.ffmpeg;
002
003import net.bramp.ffmpeg.builder.FFmpegBuilder;
004import net.bramp.ffmpeg.job.FFmpegJob;
005import net.bramp.ffmpeg.job.SinglePassFFmpegJob;
006import net.bramp.ffmpeg.job.TwoPassFFmpegJob;
007
008import java.util.List;
009import java.util.UUID;
010
011import static com.google.common.base.Preconditions.checkNotNull;
012
013public class FFmpegExecutor {
014
015  final FFmpeg ffmpeg;
016  final FFprobe ffprobe;
017
018  public FFmpegExecutor(FFmpeg ffmpeg, FFprobe ffprobe) {
019    this.ffmpeg = checkNotNull(ffmpeg);
020    this.ffprobe = checkNotNull(ffprobe);
021  }
022
023  public FFmpegJob createJob(FFmpegBuilder builder) {
024    // Single Pass
025    final List<String> args = builder.build();
026    return new SinglePassFFmpegJob(ffmpeg, args);
027  }
028
029  /**
030   * Info: https://trac.ffmpeg.org/wiki/x264EncodingGuide#twopass
031   * 
032   * @param builder
033   * @return A new two-pass FFmpegJob
034   */
035  public FFmpegJob createTwoPassJob(FFmpegBuilder builder) {
036
037    // Random prefix so multiple runs don't clash
038    String passlogPrefix = UUID.randomUUID().toString();
039
040    // Two pass
041    final boolean override = builder.getOverrideOutputFiles();
042
043    final List<String> args1 =
044        builder.setPass(1).setPassPrefix(passlogPrefix).overrideOutputFiles(true).build();
045
046    final List<String> args2 =
047        builder.setPass(2).setPassPrefix(passlogPrefix).overrideOutputFiles(override).build();
048
049    return new TwoPassFFmpegJob(ffmpeg, passlogPrefix, args1, args2);
050  }
051}
