001package net.bramp.ffmpeg;
002
003import com.google.gson.Gson;
004import com.google.gson.GsonBuilder;
005import net.bramp.commons.lang3.math.gson.FractionAdapter;
006import net.bramp.ffmpeg.gson.LowercaseEnumTypeAdapterFactory;
007import net.bramp.ffmpeg.gson.NamedBitsetAdapter;
008import net.bramp.ffmpeg.probe.FFmpegDisposition;
009import org.apache.commons.lang3.math.Fraction;
010
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014import static com.google.common.base.Preconditions.checkArgument;
015
016/**
017 * Helper class with commonly used methods
018 */
019public final class FFmpegUtils {
020
021  final static Gson gson = FFmpegUtils.setupGson();
022  final static Pattern BITRATE_REGEX = Pattern.compile("(\\d+(?:\\.\\d+)?)kbits/s");
023
024
025  FFmpegUtils() {
026    throw new AssertionError("No instances for you!");
027  }
028
029  /**
030   * Convert milliseconds to "hh:mm:ss.ms" String representation.
031   *
032   * @param milliseconds time duration in milliseconds
033   * @return time duration in human-readable format
034   */
035  public static String millisecondsToString(long milliseconds) {
036    // FIXME Negative durations are also supported.
037    // https://www.ffmpeg.org/ffmpeg-utils.html#Time-duration
038    checkArgument(milliseconds >= 0, "milliseconds must be positive");
039
040    long seconds = milliseconds / 1000;
041    milliseconds = milliseconds - (seconds * 1000);
042
043    long minutes = seconds / 60;
044    seconds = seconds - (minutes * 60);
045
046    long hours = minutes / 60;
047    minutes = minutes - (hours * 60);
048
049    if (milliseconds == 0)
050      return String.format("%02d:%02d:%02d", hours, minutes, seconds);
051
052    return String.format("%02d:%02d:%02d.%03d", hours, minutes, seconds, milliseconds);
053  }
054
055  /**
056   * Converts a string representation of bitrate to a long of bits per second
057   *
058   * @param bitrate in the form of 12.3kbits/s
059   * @return the bitrate in bits per second.
060   */
061  public static long parseBitrate(String bitrate) {
062    Matcher m = BITRATE_REGEX.matcher(bitrate);
063    if (!m.find()) {
064      throw new IllegalArgumentException("Invalid bitrate '" + bitrate + "'");
065    }
066
067    return (long) (Float.parseFloat(m.group(1)) * 1000);
068  }
069
070  static Gson getGson() {
071    return gson;
072  }
073
074  private static Gson setupGson() {
075    GsonBuilder builder = new GsonBuilder();
076
077    builder.registerTypeAdapterFactory(new LowercaseEnumTypeAdapterFactory());
078    builder.registerTypeAdapter(Fraction.class, new FractionAdapter());
079    builder.registerTypeAdapter(FFmpegDisposition.class, new NamedBitsetAdapter<>(
080        FFmpegDisposition.class));
081
082    return builder.create();
083  }
084}
