001package net.bramp.ffmpeg;
002
003import net.bramp.ffmpeg.io.ProcessUtils;
004
005import java.io.IOException;
006import java.util.concurrent.TimeUnit;
007import java.util.concurrent.TimeoutException;
008
009import static com.google.common.base.Preconditions.checkArgument;
010
011/**
012 * Helper class with commonly used methods
013 */
014public final class FFmpegUtils {
015
016  private FFmpegUtils() {}
017
018  /**
019   * Convert milliseconds to "hh:mm:ss.ms" String representation.
020   *
021   * @param milliseconds
022   * @return
023   */
024  public static String millisecondsToString(long milliseconds) {
025
026    checkArgument(milliseconds >= 0, "milliseconds must be >= 0");
027
028    long seconds = milliseconds / 1000;
029    milliseconds = milliseconds - (seconds * 1000);
030
031    long minutes = seconds / 60;
032    seconds = seconds - (minutes * 60);
033
034    long hours = minutes / 60;
035    minutes = minutes - (hours * 60);
036
037    if (milliseconds == 0)
038      return String.format("%02d:%02d:%02d", hours, minutes, seconds);
039
040    return String.format("%02d:%02d:%02d.%03d", hours, minutes, seconds, milliseconds);
041  }
042
043  public static void throwOnError(String cmd, Process p) throws IOException {
044    try {
045      // TODO In java 8 use waitFor(long timeout, TimeUnit unit)
046      if (ProcessUtils.waitForWithTimeout(p, 1, TimeUnit.SECONDS) != 0) {
047        // TODO Parse the error
048        throw new IOException(cmd + " returned non-zero exit status. Check stdout.");
049      }
050    } catch (TimeoutException e) {
051      throw new IOException("Timed out waiting for " + cmd + " to finish.");
052    }
053  }
054}
