028/**
029 * Wrapper around FFprobe
030 * 
031 * TODO ffprobe -v quiet -print_format json -show_format -show_streams mobileedge_1280x720.mp4
032 * 
033 * @author bramp
034 *
035 */
036public class FFprobe {
037
038  final static Logger LOG = LoggerFactory.getLogger(FFprobe.class);
039
040  final static String DEFAULT_PATH = MoreObjects.firstNonNull(System.getenv("FFPROBE"), "ffprobe");
041
042  final static Gson gson = FFmpegUtils.getGson();
043
044  final String path;
045
046  /**
047   * Function to run FFprobe. We define it like this so we can swap it out (during testing)
048   */
049  final ProcessFunction runFunc;
050
051  public FFprobe() {
052    this(DEFAULT_PATH, new RunProcessFunction());
053  }
054
055  public FFprobe(@Nonnull String path) {
056    this(path, new RunProcessFunction());
057  }
058
059  public FFprobe(@Nonnull ProcessFunction runFunction) {
060    this(DEFAULT_PATH, runFunction);
061  }
062
063  public FFprobe(@Nonnull String path, @Nonnull ProcessFunction runFunction) {
064    Preconditions.checkArgument(!Strings.isNullOrEmpty(path));
065    this.runFunc = checkNotNull(runFunction);
066    this.path = path;
067  }
068
069  public String getPath() {
070    return path;
071  }
072
073  private BufferedReader wrapInReader(Process p) {
074    return new BufferedReader(new InputStreamReader(p.getInputStream(), Charsets.UTF_8));
075  }
076
077  public FFmpegProbeResult probe(String mediaPath) throws IOException {
078    ImmutableList.Builder<String> args = new ImmutableList.Builder<String>();
079
080    // TODO Add:
081    // .add("--show_packets")
082    // .add("--show_frames")
083
084    // @formatter:off
085    args.add(path)
086        .add("-v", "quiet")
087        .add("-print_format", "json")
088        .add("-show_error")
089        .add("-show_format")
090        .add("-show_streams")
091        .add(mediaPath);
092    // @formatter:on
093
094    Process p = runFunc.run(args.build());
095    try {
096      Reader reader = wrapInReader(p);
097      if (LOG.isDebugEnabled()) {
098        reader = new LoggingFilterReader(reader, LOG);
099      }
100
101      FFmpegProbeResult result = gson.fromJson(reader, FFmpegProbeResult.class);
102
103      FFmpegUtils.throwOnError("ffprobe", p);
104
105      if (result == null) {
106        throw new IllegalStateException("Gson returned null, which shouldn't happen :(");
107      }
108
109      return result;
110
111    } finally {
112      p.destroy();
113    }
114  }
115}
