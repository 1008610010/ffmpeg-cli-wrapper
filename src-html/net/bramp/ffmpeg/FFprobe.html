001package net.bramp.ffmpeg;
002
003import com.google.common.base.MoreObjects;
004import com.google.common.collect.ImmutableList;
005import com.google.gson.Gson;
006import net.bramp.ffmpeg.io.LoggingFilterReader;
007import net.bramp.ffmpeg.probe.FFmpegProbeResult;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import javax.annotation.Nonnull;
012import javax.annotation.Nullable;
013import java.io.IOException;
014import java.io.Reader;
015import java.util.List;
016
017/**
018 * Wrapper around FFprobe
019 *
020 * @author bramp
021 *
022 */
023public class FFprobe extends FFcommon {
024
025  final static Logger LOG = LoggerFactory.getLogger(FFprobe.class);
026
027  final static String FFPROBE = "ffprobe";
028  final static String DEFAULT_PATH = MoreObjects.firstNonNull(System.getenv("FFPROBE"), FFPROBE);
029
030  final static Gson gson = FFmpegUtils.getGson();
031
032  public FFprobe() throws IOException {
033    this(DEFAULT_PATH, new RunProcessFunction());
034  }
035
036  public FFprobe(@Nonnull ProcessFunction runFunction) throws IOException {
037    this(DEFAULT_PATH, runFunction);
038  }
039
040  public FFprobe(@Nonnull String path) throws IOException {
041    this(path, new RunProcessFunction());
042  }
043
044  public FFprobe(@Nonnull String path, @Nonnull ProcessFunction runFunction) {
045    super(path, runFunction);
046  }
047
048  public FFmpegProbeResult probe(String mediaPath) throws IOException {
049    return probe(mediaPath, null);
050  }
051
052  /**
053   * Returns true if the binary we are using is the true ffprobe. This is to avoid conflict with
054   * avprobe (from the libav project), that some symlink to ffprobe.
055   *
056   * @return true iff this is the official ffprobe binary.
057   * @throws IOException If a I/O error occurs while executing ffprobe.
058   */
059  public boolean isFFprobe() throws IOException {
060    return version().startsWith("ffprobe");
061  }
062
063  /**
064   * Throws an exception if this is an unsupported version of ffprobe.
065   *
066   * @throws IllegalArgumentException if this is not the official ffprobe binary.
067   * @throws IOException If a I/O error occurs while executing ffprobe.
068   */
069  private void checkIfFFprobe() throws IllegalArgumentException, IOException {
070    if (!isFFprobe()) {
071      throw new IllegalArgumentException("This binary '" + path
072          + "' is not a supported version of ffprobe");
073    }
074  }
075
076  public void run(List<String> args) throws IOException {
077    checkIfFFprobe();
078    super.run(args);
079  }
080
081  // TODO Add Probe Inputstream
082  public FFmpegProbeResult probe(String mediaPath, @Nullable String userAgent) throws IOException {
083    checkIfFFprobe();
084
085    ImmutableList.Builder<String> args = new ImmutableList.Builder<String>();
086
087    // TODO Add:
088    // .add("--show_packets")
089    // .add("--show_frames")
090
091    args.add(path)
092        .add("-v", "quiet");
093
094    if (userAgent != null) {
095      args.add("-user-agent", userAgent);
096    }
097
098    args.add("-print_format", "json")
099        .add("-show_error")
100        .add("-show_format")
101        .add("-show_streams")
102        .add(mediaPath);
103
104    Process p = runFunc.run(args.build());
105    try {
106      Reader reader = wrapInReader(p);
107      if (LOG.isDebugEnabled()) {
108        reader = new LoggingFilterReader(reader, LOG);
109      }
110
111      FFmpegProbeResult result = gson.fromJson(reader, FFmpegProbeResult.class);
112
113      throwOnError(p);
114
115      if (result == null) {
116        throw new IllegalStateException("Gson returned null, which shouldn't happen :(");
117      }
118
119      return result;
120
121    } finally {
122      p.destroy();
123    }
124  }
125}
