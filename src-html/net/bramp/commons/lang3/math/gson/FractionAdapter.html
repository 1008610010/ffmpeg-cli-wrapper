001package net.bramp.commons.lang3.math.gson;
002
003import com.google.gson.TypeAdapter;
004import com.google.gson.stream.JsonReader;
005import com.google.gson.stream.JsonToken;
006import com.google.gson.stream.JsonWriter;
007import org.apache.commons.lang3.math.Fraction;
008
009import java.io.IOException;
010
011/**
012 * GSON TypeAdapter for Apache Commons Math Fraction Object
013 *
014 * @author bramp
015 *
016 */
017public class FractionAdapter extends TypeAdapter<Fraction> {
018
019  public Fraction read(JsonReader reader) throws IOException {
020    JsonToken next = reader.peek();
021
022    if (next == JsonToken.NULL) {
023      reader.nextNull();
024      return null;
025    }
026
027    if (next == JsonToken.NUMBER) {
028      return Fraction.getFraction(reader.nextDouble());
029    }
030
031    String fraction = reader.nextString();
032
033    // Ambiguous as to what 0/0 is, but FFmpeg seems to think it's zero
034    if (fraction.equals("0/0"))
035      return Fraction.ZERO;
036
037    return Fraction.getFraction(fraction);
038  }
039
040  public void write(JsonWriter writer, Fraction value) throws IOException {
041    if (value == null) {
042      writer.nullValue();
043      return;
044    }
045    writer.value(value.toProperString());
046  }
047}
